CONFIGURAÇÃO DO BANCO DE DADOS 

\connect ProgramacaoNotebook@localhost:3306

Conexão com o novo usuário:
\connect root@localhost:3306

Senha do usuário:
fr@nrobots

Verificação se o usuáro existe:
SELECT user, host FROM mysql.user;

Criação do usuário:
CREATE USER 'ProgramacaoNotebook'@'localhost' IDENTIFIED BY 'sua_senha';
GRANT ALL PRIVILEGES ON *.* TO 'ProgramacaoNotebook'@'localhost';
FLUSH PRIVILEGES;

Redefinição das senhas:
ALTER USER 'ProgramacaoNotebook'@'localhost' IDENTIFIED BY 'nova_senha';
FLUSH PRIVILEGES;

Visualizar todas as permissões do usuário:
SHOW GRANTS FOR 'ProgramacaoNotebook'@'localhost';

CREATE DATABASE article;

USE article;

CREATE TABLE prioridadesbr (
    distancia INT,
    qualidade INT,
    mercado INT,
    producaocientifica INT,
    reputacao INT,
    inovacao INT,
    publicaprivada VARCHAR(255),
    cidade VARCHAR(255)
);

CREATE TABLE prioridadesout (
    id id INT PRIMARY KEY,
    countryreq VARCHAR(100),
    sizereq VARCHAR(10),
    rankreq INT,
    focusreq VARCHAR(10),
    researchreq VARCHAR(10),
    statusreq VARCHAR(10),
    academicreputationreq DECIMAL(5,1),
    employerreputationreq DECIMAL(5,1),
    facultystudentreq DECIMAL(5,1),
    citationsfacultyreq DECIMAL(5,1),
    internationalfacultyreq DECIMAL(5,1),
    internationalstudentsreq DECIMAL(5,1),
    sustainabilityreq DECIMAL(5,1),
    employmentscorereq DECIMAL(5,1),
);
